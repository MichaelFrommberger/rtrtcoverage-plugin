447                                loc_e_msg_validity,           /* Validity */
448                                0,
449                                K_NULL_PTR,
450                                0);
451                }
452        }
453        /* Else */
454        else
455        {
456                /* Error : Invalid msg port id */
457                lbfl_mng (R_SACO_FSC_ERR_CONFIG,

460                        par_l_port_id,
461                        0,
462                        0,
463                        K_NULL_PTR,
464                        0);
465        }
466        /* Endif */
467
468        return(loc_ul_read_status);
