/* Returns K_BOOL_TRUE if all of the mandatory receptacles have been successfully connecte */


29
/* d to the required facets                                                                */


30
/* *************************************************************************************** */


31
T_boolean InteractiveTestSelector_areConnectionsSatisfied(InteractiveTestSelector *pThis)


32
{


33
    /* Start of user code */


34
	T_boolean lConnectionsSatisfied = K_BOOL_TRUE;


35
	T_uint32 lIndex;


36
	T_DisplayUnitLocation lDisplayUnitLocation =  INNER_LEFT_DISPLAY_UNIT;


37


38
	if(pThis->_softwareConfiguration != K_NULL)


39
	{


40
		lDisplayUnitLocation =  pThis->_softwareConfiguration->whereAmI(pThis->_softwareConfiguration);


41
	}


42
	else


43
	{


44
		lConnectionsSatisfied = K_BOOL_FALSE;


45
	}


46


47
	/* For each DU of the system */


48
	for(lIndex=0;lIndex < DISPLAY_UNITS_COUNT;lIndex++)


49
	{


50
		/* If Du index refer to my du */


51
		if ((lIndex != lDisplayUnitLocation) && (pThis->_systemDataReceiver[lIndex] == K_NULL))


52
		{


53
			lConnectionsSatisfied = K_BOOL_FALSE;


54
		}


55
		else


56
		{


57
			/* Nothing to do */


58
		}


59
	}


60


61
	/* IF DU is UC or LC */


62
	if((lDisplayUnitLocation == UPPER_CENTER_DISPLAY_UNIT) || (lDisplayUnitLocation == LOWER_CENTER_DISPLAY_UNIT))


63
	{


64
		/* Check if KCCU interface is available */


65
		for(lIndex=0;lIndex < K_MAX_KCCU_COUNT;lIndex++)


66
		{


67
			/* If Du index refer to my du */


68
			if (pThis->_kccuDataReceiver[lIndex] == K_NULL)


69
			{


70
				lConnectionsSatisfied = K_BOOL_FALSE;


71
			}


72
			else


73
			{


74
				/* Nothing to do */


75
			}


76
		}


77


78
		/* Check if HCE interface is available */


79
		if (K_NULL == pThis->_hceIdentificationData)


80
		{


81
			lConnectionsSatisfied = K_BOOL_FALSE;


82
		}


83
		else


84
		{


85
			/* Nothing to do */


86
		}


87
	}


88
	else


89
	{


90
		/* Nothing to do */


91
	}


92


93


94
	if ((K_NULL == pThis->_errorThrower) ||


95
		(K_NULL == pThis->_localDataReceiver) ||


96
	    (K_BOOL_TRUE == pThis->_testObserversOverflow))


97
	{


98
		lConnectionsSatisfied = K_BOOL_FALSE;


99
	}


100
	else


101
	{


102
		/* Nothing to do */


103
	}


104


105
	return lConnectionsSatisfied;


106
    /* End of user code */


107
}
